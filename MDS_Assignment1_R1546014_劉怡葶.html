# 參數設定
W0 = 15 # 關鍵在製品水準
T0 = 41 # 最小生產週期時間
rb = 0.375 # 瓶頸站產出率

# 計算CT&TH
ct = []
for wip in range(30):
    if wip <= W0:
        ct.append(T0)
    else:
        ct.append(wip / rb)
th = []
for wip in range(30):
    if wip <= W0:
        th.append(wip / T0)
    else:
        th.append(rb)
